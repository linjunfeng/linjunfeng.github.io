更简单的多维数组转码方式

最近在开发一个采集工具。数据要保存到csv。要把多维数组的utf8数据转为gbk , 第一反应是用递归去转码。后来想想var_export 可以返回一个变量的字符串 。那样就可以直接转码！

<?php

class Csv {
    public static function put($filename, $fields) {

        eval('$fields = ' . iconv("UTF-8", "GBK" . '//IGNORE', var_export($fields, TRUE)) . ';');
        $fp = fopen($filename, 'a') or die("Can't open {$filename} \n");
        while (1) {
            if (flock($fp, LOCK_EX)) {
                fputcsv($fp, $fields);
                flock($fp, LOCK_UN);
                break;
            } else {
                sleep(1);
            }
        }
        fclose($fp);
    }
}
